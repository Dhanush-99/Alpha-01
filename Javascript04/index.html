const findEvenOdd = (N, Arr) => 
{
  let evenSum=0;
  let oddSum=0;
  
  for(let i=0;i< N;i++){
    if(Arr[i]%2==0){
      evenSum=evenSum+Arr[i];
    }
    else{
      oddSum=oddSum+Arr[i];
    }
  }
  let result=[];
  result.push(evenSum);
  result.push(oddSum);
  return result;
    
};
